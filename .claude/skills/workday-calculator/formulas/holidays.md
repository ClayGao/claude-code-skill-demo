## 假日 JSON 格式

```json
{
  "2024": {
    "holidays": [
      "2024-01-01",
      "2024-02-08",
      "2024-02-09",
      "2024-02-10",
      "2024-02-11",
      "2024-02-12",
      "2024-02-13",
      "2024-02-14",
      "2024-02-28",
      "2024-04-04",
      "2024-04-05",
      "2024-06-10",
      "2024-09-17",
      "2024-10-10"
    ],
    "makeup_days": [
      "2024-02-17",
      "2024-09-14"
    ]
  }
}
```

## 使用方式

```javascript
const holidays2024 = require('./holidays-2024.json');

function isHoliday(dateStr) {
  return holidays2024.holidays.includes(dateStr);
}

function isMakeupDay(dateStr) {
  return holidays2024.makeup_days.includes(dateStr);
}
```
